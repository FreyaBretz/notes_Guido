\begin{Lemma}{lmm-bramble-hilbert}
  An LMM is consistent of order $p$ if and only if for all polynomials
  $\phi_q$ of degree $q \le p$ and $f(t,q(t)) = q'(t)$ there holds:
  \begin{gather}
    \label{eq:lmm:14}
    L_h \phi_q = 0.
  \end{gather}
\end{Lemma}
