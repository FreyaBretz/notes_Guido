\begin{Corollary}{bvp-linear}
  The linear BVP~\eqref{eq:rwa:6} has an unique solution $u(t)$ for
  arbitrary data $f(t)$ and $g$ if and only if the $d\times d$ matrix
  \begin{gather*}
    E(a) = \rwaa + \rwab \fundamental(b;a)
  \end{gather*}
  is regular.
\end{Corollary}
