\begin{Lemma}{m-matrix-inverse}
  Let $A$ be an M-matrix. If there is a vector $w$ such that for
  the vector $v=Aw$ there holds
  \begin{gather*}
    v_i \ge 1, \quad i=1,\dots,n,
  \end{gather*}
  then
  \begin{gather}
    \label{eq:fd:4}
    \norm{A^{-1}}_\infty \le \norm{w}_\infty.
  \end{gather}
\end{Lemma}
