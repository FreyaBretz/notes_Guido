\begin{Definition}{b-stability}
  A one-step method is called \define{B-stable}, if for monotonic
  initial value problems $u'=f(u)$ with arbitrary initial value $y_0$
  and $z_0$ there holds:
  \begin{equation}
    | y_1 - z_1 | \le | y_0 - z_0 |
  \end{equation}
  independent of the time step size $h$.
\end{Definition}
