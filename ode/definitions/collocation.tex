\begin{Definition}{collocation}
  An $\rks$-stage \define{collocation method} with support points
  $c_1,\ldots,c_s$ defines a \define{collocation polynomial} $y(t)\in
  \pol_s$ through
  \begin{subequations}
    \label{eq:implicit:13}
    \begin{align}
      \label{eq:implicit:2}
      y(t_0) &= y_0 \\
      \label{eq:implicit:12}
      y'(t_0+\rkc_i h) &= f\bigl(t_0+\rkc_i h, y(t_0+\rkc_i h)\bigr)
      \quad i=1,\dots,s.
    \end{align}
    We define the value at the end of the time step as
    \begin{gather}
      y_1 = y(t_0+h).
    \end{gather}
  \end{subequations}
\end{Definition}
