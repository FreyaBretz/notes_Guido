\subsection{Dreiecksmatrizen}

\begin{Definition}{dreiecksmatrix}
  Für eine \define{untere Dreiecksmatrix} $L \in \R^{n\times n}$ gilt
  \begin{gather}
    \ell_{ij} = 0,\qquad j>i.
  \end{gather}
  Für eine \define{obere Dreiecksmatrix} $R \in \R^{n\times n}$ gilt
  \begin{gather}
    \r_{ij} = 0,\qquad j<i.
  \end{gather}
\end{Definition}
